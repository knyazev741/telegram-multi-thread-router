```python
"""Typed configuration loaded from .env file."""

from pydantic_settings import BaseSettings, SettingsConfigDict


class Settings(BaseSettings):
    """Bot configuration. All fields loaded from environment or .env file."""

    bot_token: str
    owner_user_id: int
    group_chat_id: int
    auth_token: str

    model_config = SettingsConfigDict(env_file=".env", env_file_encoding="utf-8")


settings = Settings()
```

2. Create `src/__main__.py` with uvloop runner pattern (Python 3.11+):
```python
"""Entry point: python -m src"""

import asyncio
import logging

import uvloop
from aiogram import Bot, Dispatcher
from aiogram.client.default import DefaultBotProperties
from aiogram.enums import ParseMode

from src.config import settings

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
)
logger = logging.getLogger(__name__)


async def main() -> None:
    """Initialize bot and start polling."""
    bot = Bot(
        token=settings.bot_token,
        default=DefaultBotProperties(parse_mode=ParseMode.HTML),
    )

    dp = Dispatcher()

    logger.info(
        "Starting bot (owner=%d, group=%d)",
        settings.owner_user_id,
        settings.group_chat_id,
    )
    await dp.start_polling(bot)


if __name__ == "__main__":
    with asyncio.Runner(loop_factory=uvloop.new_event_loop) as runner:
        runner.run(main())
```
